Estimate autocorrelation
------------------------


.. GENERATED FROM PYTHON SOURCE LINES 36-50

.. code-block:: Python


    anaAcor = np.zeros(len(lags))
    anaAcor[0] = sigma
    estAcor = np.zeros(len(lags))
    wMu = np.mean(w)
    wSTD = np.std(w)
    for h in lags:
        if h == 0:
            estAcor[h] = 1.0
        else:
            xs = w[h:] - wMu
            xt = w[:-h] - wMu
            estAcor[h] = np.mean(xs * xt)/wSTD
